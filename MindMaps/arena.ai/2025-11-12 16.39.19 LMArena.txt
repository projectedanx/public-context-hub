class MseAdapterManager:
"""
Manages ephemeral, reversible M_se (superelastic) adapters.

Each adapter is:
- Instantiated on-demand for a specific AR node
- Trained via Test-Time Training (TTT)
- Deleted after task completion (reversibility)
"""

def __init__(self, base_model):
self.base_model = base_model
self.active_adapters = {} # {ar_node_id: adapter_instance}
self.adapter_lifecycle = {} # Audit trail

def instantiate_mse_adapter(self, ar_node_id: str, domain: str) -> LoRAAdapter:
"""
Create a new ephemeral M_se adapter for the AR node.

Args:
ar_node_id: The contradiction node ID from ALK
domain: Knowledge domain (e.g., "oncology") for scoped adaptation

Returns:
A fresh LoRA adapter with TTT capability
"""

print(f"\n⚡ M_se ADAPTER INSTANTIATION")
print(f" AR Node: {ar_node_id}")
print(f" Domain: {domain}")

# Higher rank than R-phase (more capacity for exploration)
mse_config = LoraConfig(
r=16, # Rank: 16 (vs. 4 in R-phase)
lora_alpha=32,
target_modules=["q_proj", "v_proj", "o_proj"], # Broader than R
lora_dropout=0.1, # Higher dropout for robustness

)

adapter = PeftModel(self.base_model, mse_config)
adapter.ephemeral_id = f"mse_{ar_node_id}_{int(time.time())}"
adapter.ar_node_id = ar_node_id
adapter.domain = domain
adapter.creation_timestamp = time.time()

self.adapter_lifecycle[adapter.ephemeral_id] = {

}

def destroy_adapter(self, ar_node_id: str):

Delete the M_se adapter, reverting to base model.
This is the "spring back" mechanism.

print(f"\n🗑️ M_se ADAPTER DESTRUCTION")

else:
print(f"⚠️ No active adapter for AR node {ar_node_id}")

from enum import Enum
from dataclasses import dataclass
from typing import List, Dict, Optional

class GoTNodeType(Enum):
"""Types of nodes in the Graph-of-Thoughts."""
HYPOTHESIS = "hypothesis" # A proposed explanation
EVIDENCE = "evidence" # A fact or observation
SIMULATION = "simulation" # A CoC executable test
SYNTHESIS = "synthesis" # A merge of multiple hypotheses
REFINEMENT = "refinement" # An improved version of a hypothesis

@dataclass
class GoTNode:
"""A node in the reasoning graph."""
node_id: str
node_type: GoTNodeType
content: str # Natural language or code
confidence: float = 0.0 # [0, 1]
parent_ids: List[str] = None # Nodes that spawned this
validation_status: str = "PENDING" # PENDING, VALIDATED, REJECTED
latent_embedding: np.ndarray = None # For TDA analysis

# For SYNTHESIS nodes
merged_from: List[str] = None # IDs of contradictory hypotheses

# For SIMULATION nodes
code: str = None
execution_result: Dict = None

class GoTEdgeType(Enum):
"""Types of edges in the GoT."""
SUPPORTS = "supports" # Evidence → Hypothesis
CONTRADICTS = "contradicts" # Hypothesis ↔ Hypothesis
SPAWNS = "spawns" # Parent → Child (refinement)
MERGES_INTO = "merges_into" # Multiple → Synthesis

class GraphOfThoughts:
"""
The reasoning graph engine.

Operations:
- SPAWN: Create a new hypothesis from a parent
- MERGE: Synthesize contradictory hypotheses
- REFINE: Improve a hypothesis based on feedback
- VALIDATE: Check hypothesis via CoC simulation
"""

def __init__(self, alk_kernel, mse_adapter, vw3_inductor):
self.graph = nx.DiGraph() # Directed graph
self.alk = alk_kernel
self.adapter = mse_adapter
self.vw3 = vw3_inductor
self.node_registry = {}

# LLM for hypothesis generation
self.model = mse_adapter # Uses the ephemeral adapter

def initialize_from_ar_node(self, ar_node_id: str):
"""
Seed the GoT from a contradiction (AR node).

Creates two initial HYPOTHESIS nodes from the conflicting facts.
"""

# Fetch AR node from ALK
ar_node = self.alk.graph.get_node(ar_node_id)
conflict_pair = ar_node['ar_metadata']['conflict_pair']

fact_1, fact_2 = conflict_pair

# Create hypothesis nodes for each contradictory fact
h1 = self.add_node(
node_type=GoTNodeType.HYPOTHESIS,
content=fact_1['assertion'],
confidence=fact_1['confidence'],
parent_ids=None # Root nodes
)

h2 = self.add_node(
node_type=GoTNodeType.HYPOTHESIS,
content=fact_2['assertion'],
confidence=fact_2['confidence'],
parent_ids=None
)

# Add contradiction edge
self.graph.add_edge(
h1.node_id, h2.node_id,
edge_type=GoTEdgeType.CONTRADICTS,
weight=-1.0
)

print(f"\n🌱 GoT INITIALIZED from AR node {ar_node_id}")
print(f" H1: {h1.content[:60]}...")
print(f" H2: {h2.content[:60]}...")
print(f" Edge: CONTRADICTS")

return h1, h2

def add_node(self,
node_type: GoTNodeType,
content: str,
confidence: float = 0.0,
parent_ids: List[str] = None,
**kwargs) -> GoTNode:
"""Add a new node to the reasoning graph."""

node_id = f"{node_type.value}_{len(self.node_registry)}"

node = GoTNode(
node_id=node_id,
node_type=node_type,
content=content,
confidence=confidence,
parent_ids=parent_ids or [],
**kwargs
)

self.node_registry[node_id] = node
self.graph.add_node(node_id, **node.__dict__)

# Add parent edges
if parent_ids:
for parent_id in parent_ids:
self.graph.add_edge(parent_id, node_id, edge_type=GoTEdgeType.SPAWNS)

return node

def merge_operation(self,
contradicting_node_ids: List[str],
dissonance_level: float = 0.3) -> GoTNode:
"""
The core MERGE operation: synthesize contradictory hypotheses.

This is the computational enactment of an Algorithmic Reparation (AR) node.

Args:
contradicting_node_ids: IDs of contradictory hypotheses (H1, H2, ...)
dissonance_level: VW₃ perturbation intensity [0, 1]

Returns:
A new SYNTHESIS node (e.g., H2 in the Oncology trace)
"""

print(f"\n🔀 MERGE OPERATION")
print(f" Input: {len(contradicting_node_ids)} contradictory hypotheses")
print(f" Dissonance Level: {dissonance_level}")

# 1. Fetch contradictory hypotheses
hypotheses = [self.node_registry[nid] for nid in contradicting_node_ids]

# 2. Construct synthesis prompt
synthesis_prompt = self._build_synthesis_prompt(hypotheses)

# 3. Apply VW₃ dissonance (controlled perturbation)
perturbed_prompt = self.vw3.perturb_prompt(
synthesis_prompt,
intensity=dissonance_level
)

# 4. Generate synthesis using M_se adapter
synthesis_text = self._generate_synthesis(perturbed_prompt)

# 5. Create SYNTHESIS node
synthesis_node = self.add_node(
node_type=GoTNodeType.SYNTHESIS,
content=synthesis_text,
confidence=0.5, # Tentative; will be validated by CoC
parent_ids=contradicting_node_ids,
merged_from=contradicting_node_ids
)

# 6. Add MERGES_INTO edges
for nid in contradicting_node_ids:
self.graph.add_edge(
nid, synthesis_node.node_id,
edge_type=GoTEdgeType.MERGES_INTO
)

print(f" ✓ Synthesis Node: {synthesis_node.node_id}")
print(f" Content: {synthesis_text[:80]}...")

return synthesis_node

def _build_synthesis_prompt(self, hypotheses: List[GoTNode]) -> str:
"""Construct a prompt for synthesizing contradictory hypotheses."""

prompt = "You are an antifragile reasoning system. "
prompt += "The following hypotheses appear contradictory:\n\n"

for i, h in enumerate(hypotheses):
prompt += f"H{i+1}: {h.content}\n"

prompt += "\nYour task: Synthesize a NOVEL hypothesis that resolves "
prompt += "this contradiction by operating at a higher level of abstraction. "
prompt += "Do not simply choose one hypothesis over the other. "
prompt += "Instead, identify the implicit assumption that makes them seem "
prompt += "contradictory, and propose a framework where both can be true.\n\n"
prompt += "Novel Synthesis:"

return prompt

def _generate_synthesis(self, prompt: str) -> str:
"""Generate synthesis text using the M_se adapter."""

inputs = self.adapter.tokenizer(
prompt,
return_tensors="pt",
max_length=512,
truncation=True
)

outputs = self.adapter.generate(
**inputs,
max_new_tokens=200,
temperature=0.8, # Higher temperature for creativity
top_p=0.95,
do_sample=True
)

synthesis = self.adapter.tokenizer.decode(
outputs[0],
skip_special_tokens=True
)

# Extract only the generated portion
synthesis = synthesis[len(prompt):].strip()

return synthesis
3. The Dissonance Inductor (VW₃)

class DissonanceInductor:
"""
VW₃: The "Dissonance Inductor"

Injects controlled perturbations to prevent premature convergence
and promote non-interpolative synthesis (β₁ > 0).

Mechanisms:
- Semantic perturbation (synonym substitution)
- Perspective shift (reframe from different angle)
- Conceptual crossover (inject analogy from distant domain)
"""

def __init__(self, analogy_database: Dict[str, List[str]] = None):
self.analogy_db = analogy_database or self._load_default_analogies()
self.perturbation_history = []

def perturb_prompt(self, prompt: str, intensity: float = 0.3) -> str:
"""
Apply controlled perturbations to a synthesis prompt.

Args:
prompt: Original synthesis prompt
intensity: Perturbation strength [0, 1]
- 0.0: No perturbation
- 0.3: Mild (semantic variation)
- 0.7: Strong (cross-domain analogy injection)

Returns:
Perturbed prompt
"""

if intensity == 0.0:
return prompt

perturbed = prompt

# Mechanism 1: Semantic Perturbation (always applied if intensity > 0)
if intensity >= 0.2:
perturbed = self._semantic_perturbation(perturbed, strength=intensity)

# Mechanism 2: Perspective Shift (medium intensity)
if intensity >= 0.5:
perturbed = self._perspective_shift(perturbed)

# Mechanism 3: Analogy Injection (high intensity)
if intensity >= 0.7:
perturbed = self._inject_cross_domain_analogy(perturbed, prompt)

# Log perturbation
self.perturbation_history.append({
'original': prompt,
'perturbed': perturbed,
'intensity': intensity,
'timestamp': time.time()
})

return perturbed

def _semantic_perturbation(self, text: str, strength: float) -> str:
"""
Mild perturbation: replace key terms with semantically similar variants.

Example: "resolve" → "reconcile", "unify", "bridge"
"""

# Key verbs in synthesis prompts
synonyms = {
'resolve': ['reconcile', 'unify', 'bridge', 'integrate'],
'contradiction': ['tension', 'paradox', 'dissonance', 'conflict'],
'synthesize': ['merge', 'integrate', 'unify', 'harmonize'],
'novel': ['creative', 'innovative', 'emergent', 'generative']
}

for original, variants in synonyms.items():
if original in text.lower() and random.random() < strength:
replacement = random.choice(variants)
# Case-preserving replacement
text = re.sub(
rf'\b{original}\b',
replacement,
text,
flags=re.IGNORECASE,
count=1
)

return text

def _perspective_shift(self, text: str) -> str:
"""
Reframe the synthesis task from a different perspective.
"""

perspectives = [
"Approach this as a systems theorist would:",
"Consider this from an evolutionary perspective:",
"Reframe this using information theory:",
"Analyze this through the lens of emergence:"
]

shift = random.choice(perspectives)

# Insert perspective shift before the main task
insertion_point = text.find("Your task:")
if insertion_point != -1:
text = text[:insertion_point] + shift + "\n\n" + text[insertion_point:]

return text

def _inject_cross_domain_analogy(self, text: str, original_prompt: str) -> str:
"""
Inject an analogy from a distant conceptual domain.

This is the highest-risk, highest-reward perturbation.
"""

# Extract domain from original prompt (simple keyword matching)
domain = self._detect_domain(original_prompt)

# Fetch distant analogy
if domain in self.analogy_db:
analogies = self.analogy_db[domain]
analogy = random.choice(analogies)

injection = f"\nConsider this analogy: {analogy}\n"
injection += "How might this analogy illuminate the contradiction?\n"

# Insert before "Novel Synthesis:"
insertion_point = text.find("Novel Synthesis:")
if insertion_point != -1:
text = text[:insertion_point] + injection + text[insertion_point:]

return text

def _detect_domain(self, text: str) -> str:
"""Simple domain detection via keyword matching."""

domain_keywords = {
'oncology': ['cancer', 'tumor', 'therapy', 'mutation', 'resistance'],
'physics': ['energy', 'force', 'quantum', 'particle'],
'economics': ['market', 'price', 'supply', 'demand'],
}

for domain, keywords in domain_keywords.items():
if any(kw in text.lower() for kw in keywords):
return domain

return 'general'

"""Load cross-domain analogies."""

return {

5. Test-Time Training (TTT): Adaptive M_se Updates

class TestTimeTrainer:
"""
Test-Time Training (TTT) for M_se adapter.

Updates the ephemeral adapter based on validation feedback.
"""

def __init__(self, adapter, learning_rate: float = 1e-4):
self.adapter = adapter
self.optimizer = torch.optim.AdamW(
adapter.parameters(),
lr=learning_rate
)
self.update_history = []

def update_from_validation(self,
hypothesis_node: GoTNode,
validation_result: Dict[str, Any]):
"""
Update M_se adapter weights based on validation outcome.

If validation succeeded, reinforce the adapter's current state.
If validation failed, apply corrective gradient.
"""

if validation_result['status'] == 'VALIDATED':
# Reinforcement: Self-distillation on the hypothesis
self._reinforce_hypothesis(hypothesis_node)

elif validation_result['status'] == 'REJECTED':
# Correction: Small gradient step away from rejected hypothesis
self._correct_from_rejection(hypothesis_node)

def _reinforce_hypothesis(self, hypothesis_node: GoTNode):
"""
Reinforce the adapter's parameters that generated the hypothesis.

Mechanism: Self-distillation - use the hypothesis as a training example.
"""

# Tokenize hypothesis
inputs = self.adapter.tokenizer(
hypothesis_node.content,
return_tensors="pt",
max_length=512,
truncation=True
)

# Compute loss (language modeling objective)
outputs = self.adapter(**inputs, labels=inputs['input_ids'])
loss = outputs.loss

# Backprop and update
self.optimizer.zero_grad()
loss.backward()
self.optimizer.step()

self.update_history.append({
'type': 'REINFORCE',
'hypothesis': hypothesis_node.node_id,
'loss': loss.item(),
'timestamp': time.time()
})

print(f" 🔄 TTT Update: Reinforced hypothesis (loss={loss.item():.4f})")

def _correct_from_rejection(self, hypothesis_node: GoTNode):
"""
Apply corrective update when hypothesis is rejected.

Mechanism: Inverse gradient (small step away from rejected path).
"""

# Similar to reinforcement, but invert gradient
inputs = self.adapter.tokenizer(
hypothesis_node.content,
return_tensors="pt",
max_length=512,
truncation=True
)

outputs = self.adapter(**inputs, labels=inputs['input_ids'])
loss = outputs.loss

# Invert gradient (gradient ascent, not descent)
self.optimizer.zero_grad()
loss.backward()

# Invert gradients
for param in self.adapter.parameters():
if param.grad is not None:
param.grad = -param.grad * 0.1 # Small correction factor

self.optimizer.step()

self.update_history.append({
'type': 'CORRECT',
'hypothesis': hypothesis_node.node_id,
'loss': loss.item(),
'timestamp': time.time()
})

print(f" 🔄 TTT Update: Corrected from rejection (loss={loss.item():.4f})")

class Phase1Controller:
"""
Orchestrates the full Phase 1 superelastic loop.

Workflow:
1. AR node triggers M_se instantiation
2. GoT initialized with contradictory hypotheses
3. VW₃ perturbation applied
4. GoT MERGE operation generates synthesis
5. CoC validates synthesis via simulation
6. TTT updates M_se adapter
7. Iterate until convergence or max iterations
8. Return validated synthesis to Phase 2
"""

def __init__(self,
base_model,
alk_kernel,
max_iterations: int = 5,
dissonance_level: float = 0.3):

self.alk = alk_kernel
self.max_iterations = max_iterations
self.dissonance_level = dissonance_level

# Initialize components
self.adapter_manager = MseAdapterManager(base_model)

class GraphOfThoughts:
"""
Graph-of-Thoughts engine for resolving AR nodes.
"""

def __init__(self, alk_kernel, coc_engine):
self.alk = alk_kernel # ALK from Phase 0
self.coc = coc_engine # Chain-of-Code engine
self.graph = nx.DiGraph() # Reasoning graph

def initialize_from_ar_node(self, ar_node_id: str):
"""
Seed GoT with the contradiction from ALK.
"""
# Fetch AR node metadata from ALK
ar_node = self.alk.graph.get_node(ar_node_id)

# Add core nodes to GoT
self.graph.add_node("H1", label=ar_node.conflict_pair[0])
self.graph.add_node("H2", label=ar_node.conflict_pair[1])
self.graph.add_edge("H1", "H2", relation="contradicts")

# Flag this subgraph as "active"
self.active_subgraph = ar_node_id
print(f"🌐 Initialized GoT for AR node: {ar_node_id}")

def expand_thoughts(self, max_steps: int = 5):
"""
Explore reasoning paths via BFS/Beam Search.
"""
queue = deque(["H1", "H2"]) # Start from conflicting hypotheses

for step in range(max_steps):
if not queue:
break

node = queue.popleft()

# Generate possible next steps (e.g., merge, refine)
candidates = self._generate_candidates(node)

for cand in candidates:
if cand not in self.graph:
self.graph.add_node(cand, type="hypothesis")
self.graph.add_edge(node, cand, operation="propose")
queue.append(cand)

# Optional: Prune low-energy paths
self._prune_low_energy_paths()

print(f"🔍 Expanded GoT to {len(self.graph)} nodes")

def _generate_candidates(self, node_id: str) -> List[str]:
"""
Generate new hypotheses by applying logical operations.
"""
# Example: Merge conflicting concepts
if node_id in ["H1", "H2"]: # Oncology example
return ["H2: Cycle-Targeting Therapy"]

# In practice, use LLM to propose candidates:
prompt = f"""
Given the hypothesis '{self.graph.nodes[node_id]["label"]}',
propose 3 ways to resolve contradictions with existing knowledge.
"""
response = self.model.generate(prompt) # Use M_se adapter!
hypotheses = parse_response(response)
return hypotheses

def execute_merge(self, source_node: str, target_node: str):
"""
Merge two nodes → create a new synthesis node.
"""
# 1. Create new node for synthesis
synthesis_id = f"SYN_{uuid4()}"
self.graph.add_node(
synthesis_id,
label=f"Synthesis of {source_node} and {target_node}",
source_nodes=[source_node, target_node]
)

# 2. Add edges
self.graph.add_edge(source_node, synthesis_id, operation="merge")
self.graph.add_edge(target_node, synthesis_id, operation="merge")

# 3. Flag for CoC validation
self.graph.nodes[synthesis_id]["needs_validation"] = True

print(f"🧩 Merged {source_node} → {target_node} into {synthesis_id}")
return synthesis_id
