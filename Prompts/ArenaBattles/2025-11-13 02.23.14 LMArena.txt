def __init__(self):
self.state = "EUCLIDEAN" # or PARACONSISTENT_STATE
self.propositions = {} # {p: truth_value in {T, F, B, N}}
self.provenance = {} # track evidence for each
self.contradiction_energy = 0.0
self.threshold = 0.5

def assert_proposition(self, p, evidence, source):
"""Add evidence for a proposition."""
if p not in self.propositions:
self.propositions[p] = "N" # Initially neither

# Update truth value based on evidence
if evidence > 0:
self.propositions[p] = "T" if self.propositions[p] in ["N", "T"] else "B"
else:
self.propositions[p] = "F" if self.propositions[p] in ["N", "F"] else "B"

self.provenance[p] = (evidence, source)
self._update_contradiction_energy()

def _update_contradiction_energy(self):
"""Track accumulated contradictions."""
b_count = sum(1 for v in self.propositions.values() if v == "B")
self.contradiction_energy = b_count / max(len(self.propositions), 1)

def spawn_reparation_if_needed(self):
"""Spawn AR node if contradiction energy crosses threshold."""
if self.contradiction_energy > self.threshold:
self.state = "PARACONSISTENT_STATE"
return self._get_b_state_propositions()
return None

def _get_b_state_propositions(self):
"""Return all contradictory propositions for repair."""
return [p for p, v in self.propositions.items() if v == "B"]
