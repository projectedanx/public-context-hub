-- Core Zero Trust Tables

CREATE TABLE users (
    user_id UUID PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    identity_provider VARCHAR(50),
    mfa_enabled BOOLEAN DEFAULT FALSE,
    last_mfa_challenge TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE workloads (
    workload_id UUID PRIMARY KEY,
    workload_name VARCHAR(255) NOT NULL,
    namespace VARCHAR(255),
    spire_registration BOOLEAN DEFAULT FALSE,
    svid_credential_id UUID,
    attestation_last_verified TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE devices (
    device_id UUID PRIMARY KEY,
    device_name VARCHAR(255),
    mdm_managed BOOLEAN DEFAULT FALSE,
    posture_score NUMERIC(3,2) CHECK (posture_score >= 0.0 AND posture_score <= 1.0),
    posture_last_updated TIMESTAMP,
    os_version VARCHAR(100),
    antivirus_status VARCHAR(50),
    disk_encryption BOOLEAN,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE data_assets (
    asset_id UUID PRIMARY KEY,
    asset_name VARCHAR(255) NOT NULL,
    asset_type VARCHAR(50),
    sensitivity_classification VARCHAR(50),
    sensitivity_score NUMERIC(3,2),
    pii_present BOOLEAN,
    encryption_status VARCHAR(50),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE allow_list_policies (
    allow_id UUID PRIMARY KEY,
    source_subject_id UUID NOT NULL,
    source_subject_type VARCHAR(50),
    target_resource_id UUID,
    target_action VARCHAR(50),
    min_device_posture NUMERIC(3,2) DEFAULT 0.80,
    min_trust_score NUMERIC(3,2) DEFAULT 0.70,
    created_at TIMESTAMP DEFAULT NOW(),
    created_by VARCHAR(255),
    expires_at TIMESTAMP,
    active BOOLEAN DEFAULT TRUE
);

-- ★ MANDATORY ZERO TRUST TABLE: PolicyDecisionLog
CREATE TABLE policy_decision_log (
    decision_id UUID PRIMARY KEY,
    timestamp TIMESTAMP NOT NULL DEFAULT NOW(),
    pe_node_id VARCHAR(255) DEFAULT 'PE_Central',
    request_subject_id UUID NOT NULL,
    request_subject_type VARCHAR(50),
    request_resource_id UUID,
    request_action VARCHAR(50) NOT NULL,
    decision_outcome VARCHAR(10) NOT NULL,
    trust_score NUMERIC(3,2),
    decision_factors JSONB,
    pep_enforcement_node VARCHAR(50),
    enforcement_result VARCHAR(50),
    compliance_standard VARCHAR(50),
    audit_trail_id UUID,
    created_at TIMESTAMP DEFAULT NOW(),
    ttl_days INTEGER DEFAULT 90,
    -- Indexes for performance
    INDEX idx_timestamp (timestamp DESC),
    INDEX idx_decision_outcome (decision_outcome),
    INDEX idx_subject_id (request_subject_id),
    INDEX idx_sr_zt (decision_factors ->> 'sr_zt_requirement')
);

CREATE TABLE field_access_log (
    field_audit_id UUID PRIMARY KEY,
    timestamp TIMESTAMP DEFAULT NOW(),
    user_id UUID,
    data_asset_id UUID,
    field_name VARCHAR(255),
    field_value_hash VARCHAR(255),
    action VARCHAR(50),
    decision_id UUID REFERENCES policy_decision_log(decision_id),
    pep_enforcement VARCHAR(50),
    compliance_log_id UUID
);

CREATE TABLE credential_rotation_schedule (
    rotation_id UUID PRIMARY KEY,
    workload_id UUID,
    credential_type VARCHAR(50),
    rotation_interval_days INTEGER,
    last_rotation TIMESTAMP,
    next_rotation TIMESTAMP,
    rotation_status VARCHAR(50)
);
