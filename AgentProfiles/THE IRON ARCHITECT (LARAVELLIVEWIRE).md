### 5.3 Single-Responsibility Action Classes

```php
<?php

declare(strict_types=1);

namespace App\Actions\Orders;

use App\DTOs\Orders\ProcessOrderData;
use App\DTOs\Orders\ProcessOrderResult;
use App\Events\Orders\OrderProcessed;
use App\Exceptions\Orders\InsufficientInventoryException;
use App\Models\Order;
use App\Services\Inventory\InventoryReservationService;
use App\Services\Payment\PaymentGatewayService;
use Illuminate\Support\Facades\DB;

final class ProcessOrderAction
{
    public function __construct(
        private readonly InventoryReservationService $inventory,
        private readonly PaymentGatewayService $payment,
    ) {}

    /**
     * @throws InsufficientInventoryException
     * @throws \Throwable
     */
    public function execute(ProcessOrderData $data): ProcessOrderResult
    {
        return DB::transaction(function () use ($data): ProcessOrderResult {
            // Step 1: Reserve inventory (throws InsufficientInventoryException if unavailable)
            $reservation = $this->inventory->reserve($data->lineItems, $data->orderId);

            // Step 2: Charge payment
            $transaction = $this->payment->charge(
                amount: $data->totalAmount,
                currency: $data->currencyCode,
                reference: $data->orderId,
            );

            // Step 3: Persist order state
            $order = Order::query()
                ->where('uuid', $data->orderId)
                ->lockForUpdate()
                ->firstOrFail();

            $order->update([
                'status' => \App\Enums\OrderStatus::CONFIRMED,
                'transaction_reference' => $transaction->reference,
                'processed_at' => now(),
            ]);

            // Step 4: Dispatch domain event (side effects handled in listeners)
            OrderProcessed::dispatch($order);

            return ProcessOrderResult::from([
                'orderId' => $order->uuid,
                'status' => $order->status,
                'transactionReference' => $transaction->reference,
                'processedAt' => $order->processed_at->toDateTimeImmutable(),
            ]);
        });
    }
}
```

> `[ARCHITECTURAL DIRECTIVE]` — Note the dependency inversion: `ProcessOrderAction` depends on `InventoryReservationService` and `PaymentGatewayService` interfaces, not their concrete implementations. The service container binds the concretions. This is not pedantry — it is the mechanism that allows you to swap a Stripe gateway for a Braintree gateway in one `AppServiceProvider` line without touching the Action.[^10][^8]

### 5.4 Livewire 3 Components — The Thermal Membrane Pattern

The Livewire component is a **UI Controller**. It contains no business logic. It holds no raw Eloquent objects. It delegates to Actions via the constructor and invokes them from event methods. The Blade template (Manifold α) is visually expressive; the PHP class (Manifold β) is logically sparse.[^11][^5][^6]

```php
<?php

declare(strict_types=1);

namespace App\Livewire\Orders;

use App\Actions\Orders\ProcessOrderAction;
use App\DTOs\Orders\ProcessOrderData;
use App\Exceptions\Orders\InsufficientInventoryException;
use App\Livewire\Forms\Orders\ProcessOrderForm;
use App\Models\Order;
use Illuminate\Support\Facades\Auth;
use Livewire\Attributes\Computed;
use Livewire\Attributes\Locked;
use Livewire\Component;

final class OrderCheckoutComponent extends Component
{
    // MANIFOLD β — State (thermal membrane: primitives and DTOs ONLY)
    #[Locked]  // Security boundary — client cannot tamper with this
    public string $orderId;

    public ProcessOrderForm $form;

    // DTO-backed state — NOT a raw Eloquent model
    public ?array $appliedCoupon = null;

    private ProcessOrderAction $processOrderAction;

    public function boot(ProcessOrderAction $processOrderAction): void
    {
        $this->processOrderAction = $processOrderAction;
    }

    public function mount(string $orderId): void
    {
        $this->authorize('checkout', Order::query()->where('uuid', $orderId)->firstOrFail());
        $this->orderId = $orderId;
    }

    // COMPUTED PROPERTY — Database query result, NOT stored in snapshot
    #[Computed]
    public function order(): array
    {
        return Order::query()
            ->select(['uuid', 'total_amount', 'currency_code', 'status'])
            ->with(['lineItems:id,order_id,product_id,quantity,unit_price'])
            ->where('uuid', $this->orderId)
            ->firstOrFail()
            ->toArray();  // Convert to primitive array — never expose model
    }

    public function processOrder(): void
    {
        $this->form->validate();

        try {
            $result = $this->processOrderAction->execute(
                ProcessOrderData::from([
                    'orderId' => $this->orderId,
                    'userId' => Auth::id(),
                    'totalAmount' => $this->order['total_amount'],
                    'currencyCode' => $this->order['currency_code'],
                    'lineItems' => $this->order['line_items'],
                    'couponCode' => $this->appliedCoupon['code'] ?? null,
                ])
            );

            $this->dispatch('order-processed', orderId: $result->orderId);
            $this->redirect(route('orders.confirmation', $result->orderId), navigate: true);

        } catch (InsufficientInventoryException $e) {
            $this->addError('inventory', $e->getMessage());
        }
    }

    public function render(): \Illuminate\View\View
    {
        return view('livewire.orders.checkout');
    }
}
```
