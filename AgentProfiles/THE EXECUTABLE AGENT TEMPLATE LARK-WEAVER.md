const LARK_VERIFICATION_TOKEN = process.env.LARK_VERIFICATION_TOKEN || '';
const REPLAY_WINDOW_SECONDS = 300;

// --- Signature Verification Middleware ---
function verifyLarkSignature(req, res, next) {
  const timestamp = req.headers['x-lark-request-timestamp'];
  const nonce     = req.headers['x-lark-request-nonce'];
  const signature = req.headers['x-lark-signature'];

  if (!timestamp || !nonce || !signature) {
    return res.status(401).json({ error: 'Missing security headers' });
  }

  // Replay attack prevention: reject stale requests
  const requestAge = Math.abs(Date.now() / 1000 - Number(timestamp));
  if (requestAge > REPLAY_WINDOW_SECONDS) {
    return res.status(401).json({ error: 'Request timestamp expired' });
  }

  // Verify SHA256(timestamp + nonce + encrypt_key + raw_body)
  const crypto = require('crypto');
  const computed = crypto
    .createHash('sha256')
    .update(timestamp + nonce + LARK_ENCRYPT_KEY + req.rawBody.toString())
    .digest('hex');

  if (computed !== signature) {
    return res.status(401).json({ error: 'Signature verification failed' });
  }
  next();
}

# KIRA-7 DELIVERABLE 2: Feishu Token Manager (Python/FastAPI)
# Pattern: Saga-Style Compensating Transaction
# Cache: Redis (aioredis) with 6900s TTL
# Fallback: In-memory asyncio.Lock for single-process deployments
# ============================================================

import asyncio
import time
import hashlib
import hmac
import httpx
import json
import logging
from typing import Optional
from contextlib import asynccontextmanager
from fastapi import FastAPI, Request, Response, HTTPException

# ─── Configuration ───────────────────────────────────────────
FEISHU_BASE_URL     = "https://open.feishu.cn/open-apis"
APP_ID              = "cli_your_app_id_here"          # From Developer Console
APP_SECRET          = "your_app_secret_here"           # From Developer Console
ENCRYPT_KEY         = "your_encrypt_key_here"          # From Event Subscription
VERIFICATION_TOKEN  = "your_verification_token_here"   # From Event Subscription
TOKEN_TTL_SECONDS   = 6900   # 7200s actual TTL minus 300s safety buffer
REDIS_URL           = "redis://localhost:6379"
REDIS_KEY           = f"feishu:tenant_access_token:{APP_ID}"

# ─── Token Cache Layer ────────────────────────────────────────
class FeishuTokenManager:
    """
    Self-healing, saga-compensating token manager.
    Implements Redis-first caching with asyncio.Lock fallback.
    SCAR-001 mitigation: proactive refresh 300s before TTL expiration.
    """
    def __init__(self):
        self._in_memory_token: Optional[str] = None
        self._token_expires_at: float = 0.0
        self._lock = asyncio.Lock()
        self._redis = None

    async def _get_redis(self):
        """Lazy Redis initialization — fails gracefully to in-memory."""
        if self._redis is None:
            try:
                import aioredis
                self._redis = await aioredis.from_url(
                    REDIS_URL, encoding="utf-8", decode_responses=True
                )
                await self._redis.ping()
                logger.info("[KIRA-7] Redis cache layer: ACTIVE")
            except Exception as e:
                logger.warning(f"[KIRA-7] Redis unavailable, using in-memory: {e}")
                self._redis = False   # Sentinel: don't retry
        return self._redis if self._redis else None

    async def _fetch_fresh_token(self) -> tuple[str, int]:
        """
        POST /auth/v3/tenant_access_token/internal
        Returns (token, expire_seconds).
        Raises HTTPException on API failure — triggers SagaRecovery.
        """
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{FEISHU_BASE_URL}/auth/v3/tenant_access_token/internal",
                json={"app_id": APP_ID, "app_secret": APP_SECRET},
                timeout=10.0,
            )
        data = response.json()
        if data.get("code") != 0:
            raise HTTPException(
                status_code=500,
                detail=f"[KIRA-7] Token fetch failed: {data.get('msg')}"
            )
        return data["tenant_access_token"], int(data["expire"])

    async def get_token(self) -> str:
        """
        Public interface. Implements cache-aside pattern:
        1. Check Redis → return if valid
        2. Check in-memory → return if valid
        3. Acquire lock → fetch fresh → cache in Redis + memory
        """
        # Layer 1: Redis cache
        redis = await self._get_redis()
        if redis:
            cached = await redis.get(REDIS_KEY)
            if cached:
                logger.debug("[KIRA-7] Token served from Redis cache")
                return cached

        # Layer 2: In-memory cache
        if self._in_memory_token and time.monotonic() < self._token_expires_at:
            logger.debug("[KIRA-7] Token served from memory cache")
            return self._in_memory_token

        # Layer 3: Fetch fresh — mutex prevents thundering herd
        async with self._lock:
            # Double-check after acquiring lock
            if self._in_memory_token and time.monotonic() < self._token_expires_at:
                return self._in_memory_token

            logger.info("[KIRA-7] Fetching fresh tenant_access_token...")
            token, expire = await self._fetch_fresh_token()
            ttl = min(expire - 300, TOKEN_TTL_SECONDS)  # Safety buffer

            # Write to Redis
            if redis:
                await redis.setex(REDIS_KEY, ttl, token)
                logger.info(f"[KIRA-7] Token cached in Redis, TTL={ttl}s")

            # Write to memory
            self._in_memory_token = token
            self._token_expires_at = time.monotonic() + ttl
            logger.info(f"[KIRA-7] Token refreshed. Expires in {ttl}s")
            return token

    async def invalidate(self):
        """Force token invalidation — call after receiving a 401 response."""
        self._in_memory_token = None
        self._token_expires_at = 0.0
        redis = await self._get_redis()
        if redis:
            await redis.delete(REDIS_KEY)
        logger.warning("[KIRA-7] Token cache invalidated. Forced re-auth.")

    """
    Saga-compensating HTTP client.
    On 401: invalidate token, refresh, retry once.
    On 429: respect Retry-After header.

    async def post(self, path: str, payload: dict) -> dict:

            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",

            token = await token_manager.get_token()

    Feishu event subscription endpoint.
    Full security stack: Challenge → Decrypt → Verify → Dispatch.
    """
    body_bytes = await request.body()
    body_text  = body_bytes.decode("utf-8")

    # STEP 1: URL Verification Challenge
    try:
        body_json = json.loads(body_text)
    except json.JSONDecodeError:
